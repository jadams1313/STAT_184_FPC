top_countries <- c("USA", "GBR", "URS", "GER", "KEN", "JAM", "FIN", "SWE", "CAN", "FRA")

calculate_percentage <- function(country, results1, country_data) {
  percentage <- results1 %>%
    group_by(Nationality) %>%
    filter(Nationality == country ) %>%
    summarise(Percentage = (G / sum(country_data$olympicMedalsByCountry_totGold))* 100)
  return(percentage)
}

percentage_list <- lapply(top_countries, calculate_percentage, results1 = results1, country_data = country_data)


combined_percentages <- bind_rows(percentage_list)
